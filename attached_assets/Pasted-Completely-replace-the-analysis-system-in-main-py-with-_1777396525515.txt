def score_momentum(ticker):
    score = 0
    details = {}
    
    try:
        # Fetch 12 weeks price data
        stock = yf.Ticker(ticker)
        hist = stock.history(period="3mo")
        spy = yf.Ticker("SPY").history(period="3mo")
        
        if len(hist) < 20:
            return None, {}
        
        current = hist['Close'].iloc[-1]
        week1_ago = hist['Close'].iloc[-5] if len(hist) >= 5 else hist['Close'].iloc[0]
        week4_ago = hist['Close'].iloc[-20] if len(hist) >= 20 else hist['Close'].iloc[0]
        week12_ago = hist['Close'].iloc[0]
        
        # PRICE MOMENTUM (30 points)
        ret_1w = (current - week1_ago) / week1_ago
        ret_4w = (current - week4_ago) / week4_ago
        ret_12w = (current - week12_ago) / week12_ago
        
        if ret_1w > 0: score += 10
        if ret_4w > 0: score += 10
        if ret_12w > 0: score += 10
        
        details['ret_1w'] = round(ret_1w * 100, 1)
        details['ret_4w'] = round(ret_4w * 100, 1)
        details['ret_12w'] = round(ret_12w * 100, 1)
        
        # RELATIVE STRENGTH vs SPY (20 points)
        spy_current = spy['Close'].iloc[-1]
        spy_week4_ago = spy['Close'].iloc[-20] if len(spy) >= 20 else spy['Close'].iloc[0]
        spy_ret_4w = (spy_current - spy_week4_ago) / spy_week4_ago
        
        rs = ret_4w - spy_ret_4w
        if rs > 0.03: score += 20
        elif rs > 0: score += 10
        
        details['rs_vs_spy'] = round(rs * 100, 1)
        details['spy_ret_4w'] = round(spy_ret_4w * 100, 1)
        
        # VOLUME CONFIRMATION (15 points)
        last_20 = hist.tail(20)
        up_days = last_20[last_20['Close'] > last_20['Open']]
        down_days = last_20[last_20['Close'] <= last_20['Open']]
        
        avg_up_vol = up_days['Volume'].mean() if len(up_days) > 0 else 0
        avg_down_vol = down_days['Volume'].mean() if len(down_days) > 0 else 0
        
        if avg_down_vol > 0:
            vol_ratio = avg_up_vol / avg_down_vol
            if vol_ratio > 1.5: score += 15
            elif vol_ratio > 0.8: score += 7
            details['vol_ratio'] = round(vol_ratio, 2)
        else:
            score += 15
            details['vol_ratio'] = 999
        
        # NEWS SENTIMENT (20 points)
        news_score = 0
        news_details = "No news"
        try:
            news_api_key = os.environ.get("NEWS_API_KEY")
            if news_api_key:
                url = f"https://newsapi.org/v2/everything?q={ticker}&language=en&sortBy=publishedAt&pageSize=10&apiKey={news_api_key}"
                response = requests.get(url, timeout=5)
                articles = response.json().get('articles', [])
                
                positive_words = ['beat', 'surge', 'growth', 'record', 'strong', 'upgrade', 'buy', 'bullish', 'profit', 'raised', 'exceeded', 'outperform']
                negative_words = ['miss', 'drop', 'fall', 'weak', 'downgrade', 'sell', 'bearish', 'loss', 'cut', 'below', 'concern', 'risk', 'lawsuit']
                
                pos_count = 0
                neg_count = 0
                for article in articles:
                    title = (article.get('title') or '').lower()
                    desc = (article.get('description') or '').lower()
                    text = title + ' ' + desc
                    if any(w in text for w in positive_words): pos_count += 1
                    if any(w in text for w in negative_words): neg_count += 1
                
                total = pos_count + neg_count
                if total > 0:
                    pos_ratio = pos_count / total
                    if pos_ratio >= 0.7: news_score = 20
                    elif pos_ratio >= 0.4: news_score = 10
                    else: news_score = 0
                else:
                    news_score = 10
                
                news_details = f"{pos_count} positive, {neg_count} negative"
        except:
            news_score = 10
            news_details = "Unavailable"
        
        score += news_score
        details['news'] = news_details
        
        # EARNINGS MOMENTUM (15 points)
        earnings_score = 0
        try:
            info = stock.info
            # Beat last quarter
            earnings_hist = stock.quarterly_earnings
            if earnings_hist is not None and not earnings_hist.empty:
                if 'Actual' in earnings_hist.columns and 'Estimate' in earnings_hist.columns:
                    latest = earnings_hist.iloc[0]
                    if latest['Actual'] > latest['Estimate']:
                        earnings_score += 8
                        details['earnings_beat'] = True
                    else:
                        details['earnings_beat'] = False
            
            # Estimates rising - compare current target to 30d ago approximation
            target = info.get('targetMeanPrice', 0)
            current_price = info.get('currentPrice', current)
            if target > current_price * 1.10:
                earnings_score += 7
                details['estimates_rising'] = True
            else:
                details['estimates_rising'] = False
        except:
            earnings_score = 7
            details['earnings_beat'] = None
            details['estimates_rising'] = None
        
        score += earnings_score
        
    except Exception as e:
        return None, {}
    
    return score, details
